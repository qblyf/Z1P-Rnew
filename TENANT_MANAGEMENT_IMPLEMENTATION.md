```typescript
export function validateTenantConfig(config: Partial<TenantConfig>): string[] {
  const errors: string[] = [];
  
  if (!config.id) {
    errors.push('账套ID不能为空');
  } else if (!/^[a-z0-9]+$/.test(config.id)) {
    errors.push('账套ID只能包含小写字母和数字');
  }
  
  return errors;
}
```
